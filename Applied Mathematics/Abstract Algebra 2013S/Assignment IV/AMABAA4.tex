{\bf 群, 习题7}
\begin{description}
    \item[1.]{\bf 解：}\\
    设$G=\langle a\rangle =\left\{a,a^2,\ldots ,a^n=e\right\}$,
    并设$q=\frac{n}{m}$.\\
    $\forall i,s.t.\left(a^i\right)^m=e$, 做带余除法:
    $$i={lq}+r, 0\leq r<q$$
    由于$\left(a^i\right)^m=a^{\ln +\text{rm}}=a^{{rm}}=e,0\leq {rm}<n$,
    $r=0$.\\ 这样的$i$恰好有$q,2q,\ldots ,\text{qm}=n$共$m$个. 从而原命题得证.
    \item[2.]{\bf 解：}\\
    设:$$\langle a\rangle =G\overset{\varphi }{\longrightarrow }G_1$$$$x|\rightarrow \varphi (x)$$
    $\forall y=\varphi (x),x\in G$$$y=\varphi \left(a^k\right)=(\varphi (a))^k,k\in \mathbb{N}$$
    从而$y$是$\varphi (a)$的生成元, $G$是循环群.
    \item[3.]{\bf 解：}\\
    $\left\{a,a^2,\ldots ,a^n=e\right\}=G_1\subset G\Longleftrightarrow a\in G,o(a)=n$.\\
    1)$\forall a\in G,a\neq e, o(a)| |G|=p$, 故$o(a)=p$. $G$有一个阶为$p$的循环子群. \\
    由$G$的阶为$p$, 这一循环子群就为$G$本身. 即$G$为循环群.\\
    2)$\forall a\in G,a\neq e, o(a)| |G|=2p$\\
    i)$\exists a\in G, s.t.o(a)=2p$, $G=\langle a\rangle$为交换群, 与题意矛盾！\\
    ii)$\forall a\in G, s.t. o(a)=2$. 则$a^2=e$, $G$为交换群, 与题意矛盾！\\
    iii)由以上推理, $\exists a\in G, s.t. o(a)=2$. 故$G$有一个阶为$p$的循环子群.
    \newpage
    \item[5.]{\bf 解：}\\
    1)显然$a^m\in \langle a\rangle $, 故$\left\langle a^m\right\rangle \subset \langle a\rangle$.\\
    因此$\left\langle a^m\right\rangle =\langle a\rangle $\\
    $\Longleftrightarrow \exists k\in \mathbb{N}, s.t. \left(a^m\right)^k=a$\\
    $\Longleftrightarrow {mk}\equiv 1({mod} n)$\\
    $\Longleftrightarrow \exists k,v\in \mathbb{N},s.t.{km}+{vn}=1$\\
    $\Longleftrightarrow (m,n)=1$.\\
    2)同上题, 由于$d|m$显然$\left\langle a^m\right\rangle \subset \langle a^d\rangle$.\\
    由题意, $\exists u,v\in \mathbb{N},s.t.{um}+{vn}=d$\\
    故, $$a^d=a^{d-\text{vn}}=\left(a^m\right)^u\in \left\langle a^m\right\rangle $$
    $\left\langle a^d\right\rangle \subset \left\langle a^m\right\rangle$, 故$\left\langle a^d\right\rangle =\left\langle a^m\right\rangle$.


\end{description}

{\bf 群, 习题8}
\begin{description}
    \item[1.]{\bf 解：}\\
    由题意, $H$的左(右)陪集个数为2.\\
    显然$H=eH$为$H$的一个左陪集, $H=He$为$H$的一个右陪集.\\
    任取$a\in G\cap H^C$, 则$\forall h\in H$, 否则${ah}\bar{\in }Ha={ahh}^{-1}\in H$.\\
    ${aH}\neq H$为$H$的另一个左陪集；同理${Ha}\neq H$为$H$的另一个右陪集.\\
    由陪集的性质:$${aH}=G\cap H^C={Ha}$$
    而对于任意$H$中的元素$a$, 上式显然成立. 故$H$是正规子群.
    \item[2.]{\bf 解：}\\
    $Z(G)=\left\{g\in G\left|\text{gxg}^{-1}\right.=x,\forall x\in G\right\}$\\

    $\forall z_1,z_2\in Z(G),\left(z_1z_2\right)x\left(z_1z_2\right){}^{-1}=z_1{xz}_1{}^{-1},\forall x\in G$, 故$z_1z_2\in Z(G)$\\
    $\forall z_1\in Z(G),z_1{}^{-1}{xz}_1=z_1{}^{-1}z_1{xz}_1z_1{}^{-1}=x$ ,故${z_1}^{-1}\in Z(G)$\\
    故$Z(G)$是子群. 又$\forall g\in G, {gZ}(G)g^{-1}={gg}^{-1}Z(G)=Z(G)$.\\
    故$Z(G)$是正规子群.

{\bf 群, 习题10}
\begin{description}
   \item[1.]{\bf 解：}\\
   易知${SL}_n(F)$是${GL}_n(F)$的正规子群.\\
   $\forall A,B\in {GL}_n(F)$,$$A\in {BSL}_n(F)\Longleftrightarrow \exists C\in {SL}_n(F),s.t. A=\text{CB}\Longleftrightarrow |A|=|B|$$
   这是因为, 若$\exists C\in {SL}_n(F),s.t. A={CB}$, 则$|A|=|C|\cdot |B|=|B|$;\\
   若$|A|=|B|$, 则$A=B\cdot \left(B^{-1}A\right)$, 其中$\left|B^{-1}A\right|=1$.\\
   故$\forall A,B\in {GL}_n(F)$, $A,B$属于${SL}_n(F)$的同一陪集, 当且仅当$|A|=|B|\neq 0$.\\
   $\forall r\in F^*=F\backslash \{0\}$, 取$R_r=\left(
\begin{array}{cccc}
 r &   &   &   \\
   & 1 &   &   \\
   &   & \ddots &   \\
   &   &   & 1
\end{array}
\right)_{n\times n}$.\\
    则${GL}_n(F)\left/{SL}_n\right.(F)=\left\{R_r{SL}_n(F)|r\in F^*\right\}$\\
    做映射:$${GL}_n(F)\left/{SL}_n\right.(F)\overset{\varphi }{\longrightarrow }F^*$$$$\varphi :R_r{SL}_n(F)\longrightarrow r$$
    则知其为同构.
   \item[3.]{\bf 解：}\\
   做映射$$G\overset{\pi }{\longrightarrow }\text{Aut} G$$$$\pi :g\longrightarrow \varphi _g(a)=\text{gag}^{-1},\forall a\in G$$
   $\pi \left(g_1g_2\right)=\varphi _{g_1g_2}=g_1g_2{ag}_2{}^{-1}g_1{}^{-1}=\varphi _{g_1}\left(\varphi _{g_2}(a)\right)=\pi \left(g_1\right)\pi \left(g_2\right)$;
   且$\pi (e)=\varphi _e(a)=e_{\text{Aut} G},\forall a\in G$.
   故$\pi$是群同态. 又$Z(G)=\left\{g\in G|\pi (g)=e_{\text{Aut} G}\right\}={Ker} \pi$.\\
   $$G/Z(G)\tilde{=}\pi (G)\subset \text{Aut} G$$
   且由于$\pi (G)$对运算封闭(由其同构关系知), $\pi (G)$故为$\text{Aut} G$子群.\\
   当$\text{Aut} G$为循环群时, $\text{Aut} G$子群、$G/Z(G)$亦为循环群. \\
   设$G/Z(G)=\langle x\rangle =\left\langle x^0,x^1,x^2,\ldots \right\rangle $, 则$G=\cup x^iZ(G)$\\
    $\forall a,b\in G,\exists z_1,z_2\in Z(G),s.t.a=x^iz_1,b=x^jz_2$
    $${ab}=x^iz_1x^jz_2=z_1z_2x^ix^j=z_2x^jx^iz_1=x^jz_2x^iz_1={ba}$$
    这与群$G$不可交换矛盾！故$\text{Aut} G$为非循环群.
   \newpage
   \item[4.]{\bf 解：}\\
   做映射$$\mathbb{C}^*\overset{\pi }{\longrightarrow }\mathbb{R}^+$$$$\pi :z=\text{re}^{\text{i$\theta $}}\longrightarrow r$$
   这是满同态, 且: $$r=1=e_{\mathbb{R}^+}\Longleftrightarrow |z|=1\Longleftrightarrow z\in U$$
   所以$U=\text{Ker} \pi$. 由同态基本定理:$$\left.\mathbb{C}^*\right/U\tilde{=}\mathbb{R}^+$$
\end{description}

\begin{array}{cccc}
 0 & 1 & \cdots  & 1 \\

\end{array}

其次, 首先$M_n^0(K)$对加法和数乘封闭. 设$A, B\in M_n^0(K),k\in K$:
\begin{align} \notag
\text{tr}(A+B)&=\text{tr}(A)+\text{tr}(B)=0\\ \notag
\text{tr}(\text{kA})&=k\cdot \text{tr}(A)=0
\end{align}
故$M_n^0(K)$是$M_n(K)$的一个子空间.\\
    {\bf (2):}对于$\forall A=\left(a_{\text{ij}}\right)\in M_n^t(K)$, $\left(A-\frac{t}{n}I\right)\in M_n^0(K)$.

{\bf 习题9.1}
\begin{description}
    \item[2.]{\bf 解：}\\
    {\bf (1):} $A(x)$是线性变换.\\
    对$\forall X,Y\in M_n(K), k\in K$. 有：
    \begin{align} \notag
      A(X+Y)&=\text{XA}+\text{YA}=A(X)+A(Y)\\ \notag
      A(\text{kX})&=(\text{kX})A=k(\text{XA})=\text{kA}(X)
    \end{align}
    故$A(x)$是线性映射. 又$A(x)$将$M_n(K)$映射到$M_n(K)$, 故$A(x)$是线性变换.\\
    {\bf (2):} $A(x)$是线性变换.\\
    对$\forall X,Y\in M_n(K), k\in K$. 有：
    \begin{align} \notag
      A(X+Y)=\text{BXC}+\text{BYC}=A(X)+A(Y)\\ \notag
      A(\text{kX})=B(\text{kX})C=k(\text{BXC})=\text{kA}(X)
    \end{align}
    故$A(x)$是线性映射. 又$A(x)$将$M_n(K)$映射到$M_n(K)$, 故$A(x)$是线性变换.
    \item[3.]{\bf 解：}\\
    $A(x)$是线性变换.\\
    对$\forall f(x),g(x)\in K[x], k\in K$. 有：
    \begin{align} \notag
      A(f(x)+g(x))&=f(x+a)+g(x+a)=A(f(x))+A(g(x))\\ \notag
      A(k\cdot f(x))&=k\cdot f(x+a)=k\cdot A(f(x))
    \end{align}
    故$A(x)$是线性映射. 又$A(x)$将$K[x]$映射到$K[x]$, 故$A(x)$是线性变换.
    \item[7.]{\bf 解：}\\

      A(k\cdot f(x))&=x(kf(x))=k(xf(x))=k\cdot A(f(x))

        \end{align}

    \item[8.]{\bf 解：}\\
    {\bf 必要性：}必要性显然. 若$A$可逆，则$A$是$V$上的一个同构映射. 由$\alpha _1,\alpha _2,\cdots ,\alpha _n$是$V$上的一个基立即可以得到$\text{A$\alpha $}_1,\text{A$\alpha $}_2,\cdots ,\text{A$\alpha $}_n$也是$V$上的一个基.\\
    {\bf 充分性：}若$\text{A$\alpha $}_1,\text{A$\alpha $}_2,\cdots ,\text{A$\alpha $}_n$也是$V$上的一个基, 令:
    $$\sigma \left(\sum _{i=1}^n a_i\alpha _i\right)=\sum _{i=1}^n a_i\left(\text{A$\alpha $}_i\right)$$
    则$\sigma$是$V$到自身的同构映射, 且$\sigma \left(\alpha _i\right)=\text{A$\alpha $}_i$.\\
    故$\sigma =A$. 从而$A$是$V$到自身的同构映射,$A$可逆.\\
    综上，原命题得证.

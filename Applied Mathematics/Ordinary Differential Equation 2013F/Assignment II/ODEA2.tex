{\bf 习题2-4}
\begin{description}
    \item[1(1).]{\bf 解：}\\
    令$y={ux}$，原方程可以化为：
    \begin{align}
      x\frac{\text{$\mathbf{d}$u}}{\text{$\mathbf{d}$x}}+u&=\frac{2u-1}{2-u}\\
      \frac{2-u}{u^2-1}\text{$\mathbf{d}$u}&=\frac{\text{$\mathbf{d}$x}}{x}\\
      \intertext{积分可得，}
      \ln \left|\frac{1-u}{1+u}\right|-\frac{1}{2}\ln |u^2-1|&=\ln |x|+C\\
      \intertext{代换回去可得，}
      y-x &= C(x+y)^3
    \end{align}
    \\
    \item[1(3).]{\bf 解：}\\
    令$u=x+2y$，原方程可以化为：
    \begin{align}
    &\frac{{\mathbf{d}v}}{\text{$\mathbf{d}$x}}=1+2\frac{v+1}{2v-1}\\
    &\frac{2v-1}{4v+1}\mathbf{d}v=\mathbf{d}x\\
    \intertext{积分可得，}
    &\frac{1}{2}v-\frac{3}{8}\ln |4v+1|=x+c\\
    \intertext{代换回去可得，}
    &8y-4x-3\ln |4x+8y+1|=c
    \end{align}
    \\
    \item[2(1).]{\bf 解：}\\
    令$u=x-y$，原方程可以化为：
    \begin{align}
    \frac{\text{$\mathbf{d}$u}}{\text{$\mathbf{d}$x}}=1-\cos (u)
    \end{align}
    i.当$\cos (u)\neq 1$，原方程化为
    \begin{align}
      \frac{\text{$\mathbf{d}$u}}{1-\cos  (u)}&=\text{$\mathbf{d}$x}\\
      \intertext{积分可得，}
      \frac{1}{2}\text{ctg}\left( \frac{u}{2}\right)&=x+c\\
      \cos  \left(\frac{x-y}{2}\right)=2\text{xsin}\left(\frac{x-y}{2}\right)&+\text{csin}\left(\frac{x-y}{2}\right)
    \end{align}
    ii.当$\cos (u)=1$，原方程还有解$y =x+2\text{k$\pi $}(k\in \mathbb{Z})$。
    \\
    \item[2(3).]{\bf 解：}\\
    原方程可以写作$$\frac{2\text{y$\mathbf{d}$y}}{2\text{x$\mathbf{d}$x}}=\frac{4y^2-2x^2}{x^2+y^2+3}$$
    令$u=y^2,v=x^2$，原方程可化为：
    \begin{align}
      \frac{\text{$\mathbf{d}$u}}{\text{$\mathbf{d}$v}}&=\frac{4u-2v}{u+v+3}\\
        \intertext{令$ m=u+1, n=v+2$，原方程可化为，}
\frac{\text{$\mathbf{d}$m}}{\text{$\mathbf{d}$n}}&=\frac{4m-2n}{m+n}\\
\intertext{再次代换，令$z=\frac{m}{n}$，原方程可化为：}
\frac{(z+1)\text{$\mathbf{d}$z}}{(1-z)(z-2)}&=\frac{\text{$\mathbf{d}$n}}{n}\\
\ln \left|\frac{(z-1)^2}{(2-z)^3}\right|&=C+\text{lnn}\\
\intertext{代换回来，得：}
\left(x^2-y^2+1\right)^2&=C\left(-2x^2+y^2-3\right)^3
    \end{align}

    \item[3(1).]{\bf 解：}\\
     令$z = xy$，原方程可以化为：
     \begin{align}
       \frac{\text{$\mathbf{d}$z}}{\text{$\mathbf{d}$x}}=\frac{1}{x}\left(-z^2+z-\frac{1}{4}\right)
     \end{align}
     i.$z\neq \frac{1}{2}$
     \begin{align}
       \frac{-\text{$\mathbf{d}$z}}{\left(z-\frac{1}{2}\right)^2}&=\frac{\text{$\mathbf{d}$x}}{x}\\
       \frac{1}{z-\frac{1}{2}}&=\ln |x|+C\\
       y&=\frac{1}{2x}+\frac{1}{\text{xln}|x|+\text{Cx}}
     \end{align}
     ii.$z= \frac{1}{2}$，原方程还有解$y=\frac{1}{2x}$。
     \\
    \item[4.]{\bf 解：}\\
    令$y=e^{\int u(x)dx}$，则：
    \begin{align}
      y&=e^{\int u(x)dx}\\
      y''&=u'(x)e^{\int u(x)dx}+u^2(x)e^{\int u(x)dx}\\
      \intertext{原方程可化为，}
      u'(x)e^{\int u(x)dx}&+u^2(x)e^{\int u(x)dx}+p(x)u(x)e^{\int u(x)dx}+q(x)e^{\int u(x)dx}=0\\
      \intertext{即黎卡提方程：}
      u'&=-u^2-p(x)u-q(x)
    \end{align}
\end{description}

{\bf 习题2-5}
\begin{description}
    \item[1(1).]{\bf 解：}\\
    由于：$$\frac{\frac{\partial \left(3x^2y+2\text{xy}+y^3\right)}{\partial y}-\frac{\partial \left(x^2+y^2\right)}{\partial x}}{x^2+y^2}=3$$
    积分因子$\mu (x)=\mathbf{e}^{\int 3 \, dx}=\mathbf{e}^{3x}$\\
    方程两边同乘$3\mathbf{e}^{3x}$得到，
    \begin{align}
      \left(\left(9\mathbf{e}^{3x}x^2y+6\mathbf{e}^{3x}\right)\text{xy$\mathbf{d}$x}+3\mathbf{e}^{3x}x^2\text{$\mathbf{d}$y}\right)&+\left(3\mathbf{e}^{3x}y^3\text{$\mathbf{d}$x}+3\mathbf{e}^{3x}y^2\text{$\mathbf{d}$y}\right)=0\\
      3\mathbf{e}^{3x}x^2y+\mathbf{e}^{3x}y^3&=C
    \end{align}
    \\
    \item[1(3).]{\bf 解：}\\
    方程两边同乘$xy$，则：
    \begin{align}
      \left(3x^2y+6x\right)\text{$\mathbf{d}$x}+\left(x^3+3y^2\right)\text{$\mathbf{d}$y}&=0\\
      \left(3x^2\text{y$\mathbf{d}$x}+x^3\text{$\mathbf{d}$y}\right)+\left(6\text{x$\mathbf{d}$x}+3y^2\text{$\mathbf{d}$y}\right)&=0\\
      x^3y+y^3+3x^2&=C
    \end{align}
    \\
    \item[1(5).]{\bf 解：}\\
    由于：$$\frac{\frac{\partial \left(x^2y^2-1\right)}{\partial x}-\frac{\partial \left(2\text{xy}^3\right)}{\partial y}}{2\text{xy}^3}=-\frac{2}{y}$$
    积分因子$\mu (y)=\mathbf{e}^{\int -\frac{2}{y} \, dy}=\frac{1}{y^2}$\\
    方程两边同乘$\frac{1}{y^2}$得到，
    \begin{align}
      2\text{xy$\mathbf{d}$x}+\left(x^2-\frac{1}{y^2}\right)\text{$\mathbf{d}$y}&=0\\
      \frac{1}{y}+x^2y&=C
    \end{align}

    \item[2(1).]{\bf 解：}\\
    $\mu =\mu (\phi (x,y))$是$P(x,y)\text{$\mathbf{d}$x}+Q(x,y)\text{$\mathbf{d}$y}=0$的积分因子
    $$\Longleftrightarrow \frac{\partial (\mu (\phi (x,y))P(x,y))}{\partial y}=\frac{\partial (\mu (\phi (x,y))Q(x,y))}{\partial x}$$
    $$\Longleftrightarrow \frac{\partial P}{\partial y}\mu (\phi (x,y))+\text{P$\mu $}_y(\phi (x,y))=\frac{\partial Q}{\partial x}\mu (\phi (x,y))+\text{Q$\mu $}_x(\phi (x,y))$$
    $$\Longleftrightarrow \mu (\phi (x,y))=\frac{\frac{\partial P}{\partial y}-\frac{\partial Q}{\partial x}}{Q\frac{\partial \phi }{\partial x}-P\frac{\partial \phi }{\partial x}}$$
    {\bf Q.E.D.}\\
    故，$$\frac{\frac{\partial P}{\partial y}-\frac{\partial Q}{\partial x}}{Q\mp P}=f(x\pm y)$$
    \\
    \item[3.]{\bf 解：}\\
    令$y={ux}$，设$P(x,\text{ux})\text{$\mathbf{d}$x}+Q(x,\text{ux})(\text{u$\mathbf{d}$x}+\text{x$\mathbf{d}$u})$为$m$次齐次。
    \begin{align}
      &P(x,\text{ux})\text{$\mathbf{d}$x}+Q(x,\text{ux})(\text{u$\mathbf{d}$x}+\text{x$\mathbf{d}$u})\\ \notag
      &=\left(x^mP(1,u)+\text{ux}^mQ(1,u)\right)\text{$\mathbf{d}$x}+x^{m+1}Q(1,u)\text{$\mathbf{d}$u}\\ 
      &=0 \notag
    \end{align}
    两边同时乘以$\frac{1}{\text{yQ}+\text{xP}}=\frac{1}{x^{m+1}(P(1,u)+\text{uQ}(1,u))}$，
    得到
    \begin{align}
      \frac{1}{x}\text{$\mathbf{d}$x}+\frac{Q(1,u)}{P(1,u)+\text{uQ}(1,u)}\text{$\mathbf{d}$u}=0
    \end{align}
    显然为全微分方程。
\end{description}

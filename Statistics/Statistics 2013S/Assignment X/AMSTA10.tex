\begin{description}
    \item[8.]{\bf 解：}\\
    \begin{description}
        \item[a.(2.18)]
            \begin{align}
              U&=\left\|\hat{Y}-1\overset{-}{Y}\right\|=\left\|1\hat{b}_0+X\hat{b}-1\overset{-}{X}\hat{b}-1\hat{b}_0\right\|\\
              &=\hat{b}^T\tilde{X}^T\tilde{X}\hat{b}=\hat{b}^T\tilde{X}^T\tilde{X}\left(\tilde{X}^T\tilde{X}\right)^{-1}\tilde{X}^TY\\
              &=\hat{b}^T\tilde{X}^TY=\hat{b}^T\tilde{X}^T\tilde{Y}=\hat{b}l_{\text{xy}}
            \end{align}
        \item[b.(2.19)]
            \begin{align}
              Y-1\overset{-}{Y}&=\left(Y-\hat{Y}\right)+\left(\hat{Y}-1\overset{-}{Y}\right)
              \intertext{由于$Y-\hat{Y}$垂直于子空间$\mathcal{M}(1,X)$, $\left(\hat{Y}-1\overset{-}{Y}\right)\in \mathcal{M}(1,X)$, $Y-\hat{Y}$垂直于$\left(\hat{Y}-1\overset{-}{Y}\right)$}
              \left(Y-1\overset{-}{Y}\right)^T\left(Y-1\overset{-}{Y}\right)&=\left(Y-\hat{Y}\right)^T\left(Y-\hat{Y}\right)+\left(\hat{Y}-1\overset{-}{Y}\right)^T\left(\hat{Y}-1\overset{-}{Y}\right)
              \intertext{即,}
              Q&=l_{\text{yy}}-U
            \end{align}
        \item[c.(2.20)]
            \begin{align}
             r^2&=\frac{\left(\tilde{X}^T\tilde{Y}\right)^2}{\tilde{X}^T\tilde{X}\tilde{Y}^T\tilde{Y}}=\frac{l_{\text{xy}}^2}{l_{\text{xx}}l_{\text{yy}}}=\frac{l_{\text{xy}}\cdot l_{\text{xy}}}{l_{\text{xx}}}\cdot \frac{1}{l_{\text{yy}}}=\frac{U}{l_{\text{yy}}}
              \intertext{由前一小问结论,}
              r^2&=\frac{U}{l_{\text{yy}}}=1-\frac{Q}{l_{\text{yy}}}
            \end{align}
    \end{description}
    \item[9.]{\bf 解：}\\
        \begin{description}
          \item[a.]
            \begin{align}
              \sum _{i=1}^n \frac{\hat{y}_i}{n}&=\left(1^T1\right)^{-1}1^T\hat{Y}\\
              &=\left(1^T1\right)^{-1}1^TY_{(1,X)}\\
              &=\left(1^T1\right)^{-1}1^TY_{\left(1,\tilde{X}\right)}\\
              &=\left(1^T1\right)^{-1}1^TY_1\\
              &=\overset{-}{Y}
            \end{align}
           \item[b.]
            \begin{align}
              \sum _{i=1}^n \left(\hat{y}_i-\overset{-}{y}\right)\left(y_i-\overset{-}{y}\right)&=\left(\hat{Y}-\overset{-}{Y}\right)^T\left(Y-\overset{-}{Y}\right)\\
              &=\left(\hat{Y}-1\overset{-}{Y}\right)^T\left(Y-\hat{Y}+\hat{Y}-1\overset{-}{Y}\right)\\
              &=\left(\hat{Y}-1\overset{-}{Y}\right)^T\left(\hat{Y}-1\overset{-}{Y}\right)\\
              &=U
            \end{align}
            \item[c.]
            \begin{align}
              r^2&=\frac{\left(\tilde{X}^T\tilde{Y}\right)^2}{\tilde{X}^T\tilde{X}\tilde{Y}^T\tilde{Y}}
              \intertext{由上一小问结论}
              U&=\left(\hat{Y}-\overset{-}{Y}\right)^T\left(Y-\overset{-}{Y}\right)\\
              &=\left(\hat{b}\tilde{X}\right)^T\tilde{Y}\\
              &=\left(\left(\tilde{X}^T\tilde{X}\right)^{-1}\tilde{X}^T\tilde{Y}\tilde{X}\right)^T\tilde{Y}\\
              &=\frac{\tilde{X}^TY^T\tilde{X}\tilde{Y}}{\tilde{X}^T\tilde{X}}
              \intertext{因此,}
              \frac{U}{Y^TY}&=r^2
            \end{align}
        \end{description}
